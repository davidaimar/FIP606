Como vizualizar?

df_prod <- data.frame(medias_prod)
df_prod|>
  ggplot(aes(TRAT,emmean))+
  geom_point()+
  ylim(2000, 6500)+
  geom_errorbar(aes(min = lower.CL,
                    max = upper.CL,
                    width = 0.1))+
  theme_few()+
  annotate(geom = "text", x = 1.2, y = 4250,
           label = "a")+
    annotate(geom = "text", x = 2.3, y = 4900,
           label = "ab")+
